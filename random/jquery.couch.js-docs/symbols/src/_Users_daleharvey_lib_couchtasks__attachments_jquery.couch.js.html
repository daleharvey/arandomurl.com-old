 43   function encodeDocId(docID) {
 44     var parts = docID.split("/");
 45     if (parts[0] == "_design") {
 46       parts.shift();
 47       return "_design/" + encodeURIComponent(parts.join('/'));
 48     }
 49     return encodeURIComponent(docID);
 50   }

 55   function prepareUserDoc(user_doc, new_password) {
 56     if (typeof hex_sha1 == "undefined") {
 57       alert("creating a user doc requires sha1.js to be loaded in the page");
 58       return;
 59     }
 60     var user_prefix = "org.couchdb.user:";
 61     user_doc._id = user_doc._id || user_prefix + user_doc.name;
 62     if (new_password) {
 63       // handle the password crypto
 64       user_doc.salt = $.couch.newUUID();
 65       user_doc.password_sha = hex_sha1(new_password + user_doc.salt);
 66     }
 67     user_doc.type = "user";
 68     if (!user_doc.roles) {
 69       user_doc.roles = [];
 70     }
 71     return user_doc;
 72   }

371         /**
372          * @namespace
373          * $.couch.db.changes provides an API for subscribing to the changes feed
374          * <pre><code>var $changes = $.couch.db("mydatabase").changes();
375          *$changes.onChange = function (data) {
376          *    ... process data ...
377          * }
378          * $changes.stop();
379          * </code></pre>
380          */
381         changes: function(since, options) {
382 
383           options = options || {};
384           // set up the promise object within a closure for this handler
385           var timeout = 100, db = this, active = true,
386             listeners = [],
387             promise = /** @lends $.couch.db.changes */ {
388               /**
389                * Add a listener callback
390                * @see <a href="http://techzone.couchbase.com/sites/default/files/uploads/all/documentation/couchbase-api-db.html#couchbase-api-db_db-changes_get">docs for /db/_changes</a>
391                * @param {Function} fun Callback function to run when notified of changes.
392                */
393             onChange : function(fun) {
394               listeners.push(fun);
395             },
396               /**
397                * Stop subscribing to the changes feed
398                */
399             stop : function() {
400               active = false;
401             }
402           };
403           // call each listener when there is a change
404           function triggerListeners(resp) {
405             $.each(listeners, function() {
406               this(resp);
407             });
408           };
409           // when there is a change, call any listeners, then check for another change
410           options.success = function(resp) {
411             timeout = 100;
412             if (active) {
413               since = resp.last_seq;
414               triggerListeners(resp);
415               getChangesSince();
416             };
417           };
418           options.error = function() {
419             if (active) {
420               setTimeout(getChangesSince, timeout);
421               timeout = timeout * 2;
422             }
423           };
424           // actually make the changes request
425           function getChangesSince() {
426             var opts = $.extend({heartbeat : 10 * 1000}, options, {
427               feed : "longpoll",
428               since : since
429             });
430             ajax(
431               {url: db.uri + "_changes"+encodeOptions(opts)},
432               options,
433               "Error connecting to "+db.uri+"/_changes."
434             );
435           }
436           // start the first request
437           if (since) {
438             getChangesSince();
439           } else {
440             db.info({
441               success : function(info) {
442                 since = info.update_seq;
443                 getChangesSince();
444               }
445             });
446           }
447           return promise;
448         },

556         /**
557          * Create a new document in the specified database, using the supplied JSON document structure. If the JSON structure includes the _id field, then the document will be created with the specified document ID. If the _id field is not specified, a new unique ID will be generated.
558          * @see <a href="http://techzone.couchbase.com/sites/default/files/uploads/all/documentation/couchbase-api-dbdoc.html#couchbase-api-dbdoc_db_post">docs for GET /db/doc</a>
559          * @param {String} doc document to save
560          * @param {ajaxSettings} options <a href="http://api.jquery.com/jQuery.ajax/#jQuery-ajax-settings">jQuery ajax settings</a>
561          */
562         saveDoc: function(doc, options) {
563           options = options || {};
564           var db = this;
565           var beforeSend = fullCommit(options);
566           if (doc._id === undefined) {
567             var method = "POST";
568             var uri = this.uri;
569           } else {
570             var method = "PUT";
571             var uri = this.uri + encodeDocId(doc._id);
572           }
573           var versioned = maybeApplyVersion(doc);
574           $.ajax({
575             type: method, url: uri + encodeOptions(options),
576             contentType: "application/json",
577             dataType: "json", data: toJSON(doc),
578             beforeSend : beforeSend,
579             complete: function(req) {
580               var resp = httpData(req, "json");
581               if (req.status == 200 || req.status == 201 || req.status == 202) {
582                 doc._id = resp.id;
583                 doc._rev = resp.rev;
584                 if (versioned) {
585                   db.openDoc(doc._id, {
586                     attachPrevRev : true,
587                     success : function(d) {
588                       doc._attachments = d._attachments;
589                       if (options.success) options.success(resp);
590                     }
591                   });
592                 } else {
593                   if (options.success) options.success(resp);
594                 }
595               } else if (options.error) {
596                 options.error(req.status, resp.error, resp.reason);
597               } else {
598                 alert("The document could not be saved: " + resp.reason);
599               }
600             }
601           });
602         },

878   var httpData = $.httpData || function( xhr, type, s ) { // lifted from jq1.4.4
879     var ct = xhr.getResponseHeader("content-type") || "",
880       xml = type === "xml" || !type && ct.indexOf("xml") >= 0,
881       data = xml ? xhr.responseXML : xhr.responseText;
882 
883     if ( xml && data.documentElement.nodeName === "parsererror" ) {
884       $.error( "parsererror" );
885     }
886     if ( s && s.dataFilter ) {
887       data = s.dataFilter( data, type );
888     }
889     if ( typeof data === "string" ) {
890       if ( type === "json" || !type && ct.indexOf("json") >= 0 ) {
891         data = $.parseJSON( data );
892       } else if ( type === "script" || !type && ct.indexOf("javascript") >= 0 ) {
893         $.globalEval( data );
894       }
895     }

902   function ajax(obj, options, errorMessage, ajaxOptions) {
903     options = $.extend({successStatus: 200}, options);
904     ajaxOptions = $.extend({contentType: "application/json"}, ajaxOptions);
905     errorMessage = errorMessage || "Unknown error";
906     $.ajax($.extend($.extend({
907       cache:false,
908       type: "GET",
909       dataType: "json",
910       beforeSend: function(xhr){
911         if(ajaxOptions && ajaxOptions.headers){
912           for (var header in ajaxOptions.headers){
913             xhr.setRequestHeader(header, ajaxOptions.headers[header]);
914           }
915         }
916       },
917       complete: function(req) {
918         try {
919           var resp = httpData(req, "json");
920         } catch(e) {
921           if (options.error) {
922             options.error(req.status, req, e);
923           } else {
924             alert(errorMessage + ": " + e);
925           }
926           return;
927         }
928         if (options.ajaxStart) {
929           options.ajaxStart(resp);
930         }
931         if (req.status == options.successStatus) {
932           if (options.beforeSuccess) options.beforeSuccess(req, resp);
933           if (options.success) options.success(resp);
934         } else if (options.error) {
935           options.error(req.status, resp && resp.error || errorMessage,
936                         resp && resp.reason || "no response");
937         } else {
938           alert(errorMessage + ": " + resp.reason);
939         }
940       }
941     }, obj), ajaxOptions));
942   }

962   // Convert a options object to an url query string.
963   // ex: {key:'value',key2:'value2'} becomes '?key="value"&key2="value2"'
964   function encodeOptions(options) {
965     var buf = [];
966     if (typeof(options) === "object" && options !== null) {
967       for (var name in options) {
968         if ($.inArray(name, ["error", "success", "beforeSuccess", "ajaxStart"]) >= 0)
969           continue;
970         var value = options[name];
971         if ($.inArray(name, ["key", "startkey", "endkey"]) >= 0) {
972           value = toJSON(value);
973         }
974         buf.push(encodeURIComponent(name) + "=" + encodeURIComponent(value));
975       }
976     }
977     return buf.length ? "?" + buf.join("&") : "";
978   }
